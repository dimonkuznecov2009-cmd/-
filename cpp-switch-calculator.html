📄 calculator.cpp 
// Калькулятор на C++ с оператором switch
#include <iostream>
#include <cmath>
#include <limits>

using namespace std;

// Функция для очистки потока ввода после ошибки
void clearInput() {
    cin.clear();
    cin.ignore(numeric_limits<streamsize>::max(), '\n');
}

// Функция для получения числа с проверкой
double getNumber(const string& prompt) {
    double num;
    while (true) {
        cout << prompt;
        cin >> num;
        
        if (cin.fail()) {
            cout << "Ошибка! Пожалуйста, введите число.\n";
            clearInput();
        } else {
            break;
        }
    }
    return num;
}

// Функция для получения оператора с проверкой
char getOperator() {
    char op;
    while (true) {
        cout << "Выберите операцию (+, -, *, /, ^, s - квадрат, k - корень): ";
        cin >> op;
        
        if (op == '+' || op == '-' || op == '*' || op == '/' || 
            op == '^' || op == 's' || op == 'k') {
            break;
        } else {
            cout << "Неверный оператор! Попробуйте снова.\n";
        }
    }
    return op;
}

// Функция для выполнения вычислений
void calculate() {
    cout << "\n========================================\n";
    cout << "       ИНЖЕНЕРНЫЙ КАЛЬКУЛЯТОР\n";
    cout << "========================================\n\n";
    
    char choice;
    
    do {
        // Получаем первый операнд
        double a = getNumber("Введите первое число: ");
        
        // Получаем оператор
        char op = getOperator();
        
        double b = 0;
        double result = 0;
        bool valid = true;
        
        // Унарные операции (квадрат, корень) не требуют второго числа
        if (op == 's' || op == 'k') {
            switch (op) {
                case 's':
                    result = a * a;
                    cout << "\n" << a << "² = " << result << endl;
                    break;
                case 'k':
                    if (a < 0) {
                        cout << "\nОшибка! Нельзя извлечь корень из отрицательного числа.\n";
                        valid = false;
                    } else {
                        result = sqrt(a);
                        cout << "\n√" << a << " = " << result << endl;
                    }
                    break;
            }
        } else {
            // Бинарные операции требуют второго числа
            b = getNumber("Введите второе число: ");
            
            switch (op) {
                case '+':
                    result = a + b;
                    break;
                case '-':
                    result = a - b;
                    break;
                case '*':
                    result = a * b;
                    break;
                case '/':
                    if (b == 0) {
                        cout << "\nОшибка! Деление на ноль невозможно.\n";
                        valid = false;
                    } else {
                        result = a / b;
                    }
                    break;
                case '^':
                    result = pow(a, b);
                    break;
                default:
                    cout << "\nНеизвестная операция!\n";
                    valid = false;
            }
            
            if (valid) {
                cout << "\n" << a << " " << op << " " << b << " = " << result << endl;
            }
        }
        
        if (valid && op != 's' && op != 'k') {
            cout << "\nРезультат: " << result << endl;
        }
        
        cout << "\n----------------------------------------\n";
        cout << "Продолжить вычисления? (y/n): ";
        cin >> choice;
        cout << endl;
        
    } while (choice == 'y' || choice == 'Y');
    
    cout << "\nСпасибо за использование калькулятора!\n";
}

int main() {
    calculate();
    return 0;
}

🔍 Разбор кода шаг за шагом

1

Подключение библиотек и пространство имён

#include <iostream>
#include <cmath>
#include <limits>

using namespace std;

Что здесь происходит:

  • #include <iostream> — для ввода/вывода (cin, cout)
  • #include <cmath> — для математических функций (sqrt, pow)
  • #include <limits> — для работы с лимитами типов (нужно для очистки ввода)
  • using namespace std — позволяет не писать std:: перед cin, cout
2

Функция очистки потока ввода

void clearInput() {
    cin.clear();  // Сбрасываем флаги ошибок
    cin.ignore(numeric_limits<streamsize>::max(), '\n'); // Очищаем буфер
}

Что здесь происходит:

  • Если пользователь ввёл не число, cin.fail() становится true
  • cin.clear() — сбрасывает состояние ошибки
  • cin.ignore(...) — удаляет неправильные данные из буфера ввода
  • Это позволяет программе запросить ввод снова без зависания
3

Функция получения числа с проверкой

double getNumber(const string& prompt) {
    double num;
    while (true) {
        cout << prompt;
        cin >> num;
        
        if (cin.fail()) {
            cout << "Ошибка! Пожалуйста, введите число.\n";
            clearInput();
        } else {
            break;
        }
    }
    return num;
}

Что здесь происходит:

  • while (true) — бесконечный цикл, пока не будет введено корректное число
  • cin.fail() — проверяет, произошла ли ошибка ввода
  • При ошибке — выводим сообщение и очищаем ввод
  • При успехе — выходим из цикла и возвращаем число
4

Функция выбора оператора (основной switch)

switch (op) {
    case '+':
        result = a + b;
        break;
    case '-':
        result = a - b;
        break;
    case '*':
        result = a * b;
        break;
    case '/':
        if (b == 0) {
            cout << "Ошибка! Деление на ноль!\n";
            valid = false;
        } else {
            result = a / b;
        }
        break;
    case '^':
        result = pow(a, b);
        break;
    default:
        cout << "Неизвестная операция!\n";
        valid = false;
}

Что здесь происходит:

  • switch (op) — выбирает блок кода в зависимости от значения переменной op
  • case '+' — если op равно '+', выполняется сложение
  • break — выходит из switch, без него выполнение "провалится" в следующий case
  • default — выполняется, если ни один case не подошёл
  • Для деления — дополнительная проверка на ноль
  • pow(a, b) — возведение a в степень b (требует #include <cmath>)
5

Унарные операции: квадрат и корень

if (op == 's' || op == 'k') {
    switch (op) {
        case 's':
            result = a * a;
            cout << a << "² = " << result << endl;
            break;
        case 'k':
            if (a < 0) {
                cout << "Ошибка! Корень из отрицательного числа!\n";
                valid = false;
            } else {
                result = sqrt(a);
                cout << "√" << a << " = " << result << endl;
            }
            break;
    }
}

Что здесь происходит:

  • Унарные операции (квадрат 's' и корень 'k') не требуют второго числа
  • a * a — возведение в квадрат (можно также pow(a, 2))
  • sqrt(a) — квадратный корень (только для неотрицательных чисел)
  • Для корня проверяем, что число не отрицательное
6

Главный цикл do-while

do {
    // вычисления...
    cout << "Продолжить вычисления? (y/n): ";
    cin >> choice;
} while (choice == 'y' || choice == 'Y');

Что здесь происходит:

  • do-while — цикл с пост-условием (выполнится хотя бы один раз)
  • После каждого вычисления спрашиваем, хочет ли пользователь продолжить
  • Если введено 'y' или 'Y' — цикл повторяется
  • Если 'n' или другой символ — программа завершается
7

Функция main и запуск

int main() {
    calculate();
    return 0;
}

Что здесь происходит:

  • int main() — точка входа в программу
  • Вызываем функцию calculate(), где содержится вся логика калькулятора
  • return 0 — сообщает операционной системе, что программа завершилась успешно

⭐ Лёгкое

Добавление операции остатка от деления

Добавьте операцию % (остаток от деления). Используйте только для целых чисел.

Подсказка

Добавьте case '%': result = (int)a % (int)b; и не забудьте добавить '%' в меню выбора оператора.

⭐ Лёгкое

Меню с индексами

Вместо ввода символов операторов, предложите пользователю выбрать номер операции из меню.

Подсказка

Выведите меню: "1. Сложение, 2. Вычитание..." и используйте switch для чисел.

⭐⭐ Среднее

Калькулятор с памятью

Добавьте переменную "память" (M+ — добавить в память, M- — вычесть, MR — показать память, MC — очистить).

Подсказка

Объявите переменную memory = 0 и добавьте соответствующие операции в switch.

⭐⭐ Среднее

История операций

Сохраняйте все выполненные операции в массив и выводите историю по запросу пользователя.

Подсказка

Используйте vector<string> для хранения строк с операциями.

⭐⭐⭐ Сложное

Поддержка тригонометрических функций

Добавьте операции sin, cos, tan для работы с углами в градусах или радианах.

Подсказка

Используйте sin(), cos(), tan() из библиотеки cmath. Для перевода градусов в радианы: радианы = градусы * M_PI / 180.

Обработка выражений

Модифицируйте калькулятор, чтобы он принимал выражения целиком (например "25+7*3") с учётом приоритета операций.

Подсказка

Это сложная задача! Потребуется разбить строку на токены и реализовать алгоритм сортировочной станции.

▶️ Как скомпилировать и запустить

1

Установите компилятор C++

Windows: Установите MinGW-w64 или используйте Visual Studio. macOS: Установите Xcode Command Line Tools: xcode-select --install. Linux: Установите g++: sudo apt install g++ (Ubuntu/Debian).

2

Сохраните код в файл

Скопируйте код из раздела выше и вставьте в текстовый редактор. Сохраните файл с именем calculator.cpp.

3

Скомпилируйте программу

Откройте терминал/командную строку в папке с файлом и выполните:

g++ calculator.cpp -o calculator -lm

Флаг -lm подключает математическую библиотеку (нужна для pow и sqrt).

4

Запустите программу

Windows: calculator.exe
macOS/Linux: ./calculator

Также можно использовать Replit или OnlineGDB для запуска в браузере без установки.

💡 Важные моменты об операторе switch

⚠️

Не забывайте break!

Без break выполнение "проваливается" в следующий case. Иногда это полезно, но чаще приводит к ошибкам.

📋

Case работает только с целыми типами

Switch можно использовать с int, char, enum. Нельзя с double, string.

🎯

Default — на все случаи жизни

Всегда добавляйте default для обработки непредвиденных значений.

🔄

Сравнение с if-else

Switch эффективнее и читаемее при проверке одной переменной на множество значений. If-else лучше для сложных условий.
