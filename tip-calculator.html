📝 Полный код программы

📄 tip_calculator.py 
# Калькулятор чаевых на Python
def calculate_tip():
    print("=" * 40)
    print("     КАЛЬКУЛЯТОР ЧАЕВЫХ")
    print("=" * 40)
    
    # Ввод суммы счета с проверкой
    while True:
        try:
            bill = float(input("Введите сумму счета (₽): "))
            if bill <= 0:
                print("Сумма должна быть положительной!")
                continue
            break
        except ValueError:
            print("Ошибка! Пожалуйста, введите число.")
    
    # Выбор процента чаевых
    print("\nВыберите размер чаевых:")
    print("1. 10% - Обычный")
    print("2. 15% - Хороший")
    print("3. 20% - Отличный")
    print("4. Свой вариант")
    
    while True:
        choice = input("\nВаш выбор (1-4): ")
        
        if choice == '1':
            tip_percent = 10
            break
        elif choice == '2':
            tip_percent = 15
            break
        elif choice == '3':
            tip_percent = 20
            break
        elif choice == '4':
            while True:
                try:
                    tip_percent = float(input("Введите свой процент (%): "))
                    if tip_percent < 0:
                        print("Процент не может быть отрицательным!")
                        continue
                    break
                except ValueError:
                    print("Пожалуйста, введите число!")
            break
        else:
            print("Неверный выбор! Выберите 1, 2, 3 или 4.")
    
    # Расчёт чаевых и итоговой суммы
    tip_amount = bill * tip_percent / 100
    total = bill + tip_amount
    
    # Вывод результата
    print("\n" + "=" * 40)
    print("         РЕЗУЛЬТАТ")
    print("=" * 40)
    print(f"Сумма счета:      {bill:.2f} ₽")
    print(f"Чаевые ({tip_percent}%): {tip_amount:.2f} ₽")
    print(f"ИТОГО К ОПЛАТЕ:   {total:.2f} ₽")
    print("=" * 40)
    
    # Дополнительно: разделение на гостей
    split = input("\nРазделить счёт на несколько гостей? (да/нет): ").lower()
    if split in ['да', 'lf', 'yes', 'y', 'д']:
        while True:
            try:
                people = int(input("Количество гостей: "))
                if people <= 0:
                    print("Количество должно быть положительным!")
                    continue
                per_person = total / people
                print(f"Каждый должен заплатить: {per_person:.2f} ₽")
                break
            except ValueError:
                print("Пожалуйста, введите целое число!")
    
    print("\nСпасибо за использование калькулятора чаевых!")

if __name__ == "__main__":
    calculate_tip()

🔍 Разбор кода шаг за шагом

1

Определение функции

def calculate_tip():

Мы создаём функцию calculate_tip(), которая содержит всю логику программы. Это позволяет легко запускать программу и использовать её в других проектах.

2

Приветствие и ввод суммы счета

print("=" * 40)
print("     КАЛЬКУЛЯТОР ЧАЕВЫХ")
print("=" * 40)

while True:
    try:
        bill = float(input("Введите сумму счета (₽): "))
        if bill <= 0:
            print("Сумма должна быть положительной!")
            continue
        break
    except ValueError:
        print("Ошибка! Пожалуйста, введите число.")

Что здесь происходит:

  • print("=" * 40) — выводит линию из 40 символов "=" для красивого оформления
  • while True — бесконечный цикл, который повторяется, пока не будет введено корректное значение
  • try/except — перехватывает ошибку, если пользователь ввел не число
  • float(input(...)) — преобразует введённую строку в число с плавающей точкой
  • if bill <= 0 — проверяет, что сумма положительная
  • break — выходит из цикла, когда всё корректно

4

Расчёт чаевых и итога

tip_amount = bill * tip_percent / 100
total = bill + tip_amount

Формулы:

  • Сумма чаевых = сумма счета × процент чаевых ÷ 100
  • Итоговая сумма = сумма счета + чаевые

Пример: счёт 1250 ₽, чаевые 15% → чаевые = 1250 × 15 ÷ 100 = 187.5 ₽, итог = 1250 + 187.5 = 1437.5 ₽

Что здесь происходит:

  • {bill:.2f} — форматирование числа с двумя знаками после запятой
  • \n — символ новой строки для отступа перед результатом

  • int(input(...)) — преобразует ввод в целое число

❓ Часто задаваемые вопросы

Почему программа не запускается?
Проверьте: 1) Установлен ли Python. 2) Правильно ли сохранено расширение .py. 3) Нет ли ошибок в коде (например, пропущена скобка). 4) Попробуйте запустить через терминал.
Что значит .2f в f-строке?
Это формат вывода числа с плавающей точкой: .2f означает "два знака после запятой". Например, 1250.5555 → 1250.56.
Зачем нужен while True и break?
While True создаёт бесконечный цикл. Break выходит из него, когда условие выполнено. Это нужно, чтобы повторять запрос до тех пор, пока пользователь не введёт корректные данные.
Как добавить новые валюты?
Добавьте переменную для валюты и используйте её в print(). Например: currency = input("Валюта (₽/$/€): ") и затем print(f"... {total:.2f} {currency}").
