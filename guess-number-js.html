⚡ script.js 
// Игра "Угадай число" на JavaScript
class GuessNumberGame {
    constructor() {
        this.secretNumber = null;
        this.attempts = 0;
        this.bestScore = localStorage.getItem('bestScore') || null;
        this.history = [];
        this.isGameOver = false;
        
        this.initDOM();
        this.newGame();
        this.addEventListeners();
        this.updateUI();
    }
    
    initDOM() {
        this.guessInput = document.getElementById('guessInput');
        this.checkBtn = document.getElementById('checkBtn');
        this.resetBtn = document.getElementById('resetBtn');
        this.messageDiv = document.getElementById('message');
        this.attemptsSpan = document.getElementById('attempts');
        this.bestScoreSpan = document.getElementById('bestScore');
        this.historyDiv = document.getElementById('history');
    }
    
    newGame() {
        this.secretNumber = Math.floor(Math.random() * 100) + 1;
        this.attempts = 0;
        this.history = [];
        this.isGameOver = false;
        this.guessInput.disabled = false;
        this.checkBtn.disabled = false;
        this.guessInput.value = '';
        this.messageDiv.innerHTML = '🤔 Я загадал число от 1 до 100. Попробуй угадать!';
        this.updateUI();
        this.updateHistory();
    }
    
    checkGuess() {
        if (this.isGameOver) {
            this.messageDiv.innerHTML = '🎮 Игра окончена! Нажми "Новая игра" чтобы продолжить.';
            return;
        }
        
        const guess = parseInt(this.guessInput.value);
        
        if (isNaN(guess) || guess < 1 || guess > 100) {
            this.messageDiv.innerHTML = '⚠️ Пожалуйста, введите число от 1 до 100!';
            return;
        }
        
        this.attempts++;
        this.history.push(guess);
        
        if (guess === this.secretNumber) {
            this.gameWin(guess);
        } else if (guess < this.secretNumber) {
            this.messageDiv.innerHTML = `📈 ${guess} — слишком мало! Попробуй еще.`;
        } else {
            this.messageDiv.innerHTML = `📉 ${guess} — слишком много! Попробуй еще.`;
        }
        
        this.updateUI();
        this.updateHistory();
        this.guessInput.value = '';
        this.guessInput.focus();
    }
    
    gameWin(guess) {
        this.isGameOver = true;
        this.guessInput.disabled = true;
        this.checkBtn.disabled = true;
        
        if (!this.bestScore || this.attempts < this.bestScore) {
            this.bestScore = this.attempts;
            localStorage.setItem('bestScore', this.bestScore);
            this.messageDiv.innerHTML = `🎉 ПОБЕДА! Ты угадал число ${this.secretNumber} за ${this.attempts} попыток! 🎉
🏆 НОВЫЙ РЕКОРД! 🏆`;
        } else {
            this.messageDiv.innerHTML = `🎉 ПОБЕДА! Ты угадал число ${this.secretNumber} за ${this.attempts} попыток! 🎉`;
        }
        
        this.updateUI();
    }
    
    updateUI() {
        this.attemptsSpan.textContent = this.attempts;
        this.bestScoreSpan.textContent = this.bestScore ? this.bestScore : '—';
    }
    
    updateHistory() {
        this.historyDiv.innerHTML = '';
        this.history.forEach((guess, index) => {
            const item = document.createElement('div');
            item.className = 'history-item';
            let emoji = '❓';
            if (guess === this.secretNumber && this.isGameOver) {
                emoji = '✅';
            } else if (guess < this.secretNumber) {
                emoji = '📈';
            } else {
                emoji = '📉';
            }
            item.textContent = `${emoji} Попытка ${index + 1}: ${guess}`;
            this.historyDiv.appendChild(item);
        });
        this.historyDiv.scrollTop = this.historyDiv.scrollHeight;
    }
    
    addEventListeners() {
        this.checkBtn.addEventListener('click', () => this.checkGuess());
        this.resetBtn.addEventListener('click', () => this.newGame());
        this.guessInput.addEventListener('keypress', (e) => {
            if (e.key === 'Enter') this.checkGuess();
        });
    }
}

// Запуск игры
document.addEventListener('DOMContentLoaded', () => {
    new GuessNumberGame();
});

🔍 Разбор кода шаг за шагом

1

Класс игры и конструктор

class GuessNumberGame {
    constructor() {
        this.secretNumber = null;
        this.attempts = 0;
        this.bestScore = localStorage.getItem('bestScore') || null;
        this.history = [];
        this.isGameOver = false;
    }
}

Что здесь происходит:

  • class GuessNumberGame — создаём класс, который содержит всю логику игры
  • constructor() — метод, который вызывается при создании нового экземпляра игры
  • this.secretNumber — загаданное число (будет установлено позже)
  • this.attempts — счётчик попыток
  • localStorage.getItem('bestScore') — загружаем лучший результат из памяти браузера
  • this.history — массив для хранения всех попыток
  • this.isGameOver — флаг, закончена ли игра
2

Генерация случайного числа

newGame() {
    this.secretNumber = Math.floor(Math.random() * 100) + 1;
    this.attempts = 0;
    this.history = [];
    this.isGameOver = false;
}

Что здесь происходит:

  • Math.random() — генерирует случайное число от 0 (включительно) до 1 (не включительно)
  • Math.random() * 100 — даёт число от 0 до 99.999...
  • Math.floor() — округляет вниз до целого числа (0-99)
  • + 1 — смещает диапазон до 1-100
  • Сбрасываем счётчик попыток, историю и флаг окончания игры
3

Обработка ввода и проверка числа

checkGuess() {
    const guess = parseInt(this.guessInput.value);
    
    if (isNaN(guess) || guess < 1 || guess > 100) {
        this.messageDiv.innerHTML = '⚠️ Введите число от 1 до 100!';
        return;
    }
    
    this.attempts++;
    this.history.push(guess);
}

Что здесь происходит:

  • parseInt() — преобразует строку из поля ввода в число
  • isNaN(guess) — проверяет, является ли введённое значение числом
  • Проверяем, что число в диапазоне от 1 до 100
  • return — прерывает выполнение функции, если данные некорректны
  • Увеличиваем счётчик попыток и добавляем число в историю

Победа и рекорды

  • !this.bestScore — проверяем, есть ли сохранённый рекорд
  • Если текущее количество попыток меньше рекорда — сохраняем новый рекорд
  • localStorage.setItem() — сохраняем рекорд в памяти браузера (даже после перезагрузки страницы)

Обновление интерфейса и истории

📚 Задания для самостоятельной работы

Попробуйте улучшить игру самостоятельно

⭐ Лёгкое

Подсказка "Холодно-горячо"

Добавьте цветовую индикацию подсказок. Если число очень близко (разница меньше 5), показывайте "Горячо! 🔥", если далеко — "Холодно ❄️".

Подсказка

Вычислите разницу: const diff = Math.abs(guess - this.secretNumber); и используйте условия if (diff <= 5).

⭐ Лёгкое

Кнопка "Сдаться"

Добавьте кнопку "Сдаться", которая показывает загаданное число и завершает игру без победы.

Подсказка

Создайте метод surrender(), который показывает число и блокирует ввод.

⭐⭐ Среднее

Выбор уровня сложности

Добавьте выбор диапазона чисел: лёгкий (1-50), средний (1-100), сложный (1-200).

Подсказка

Добавьте <select> с вариантами и передавайте максимальное число в newGame(max).

⭐⭐ Среднее

Анимация при победе

Добавьте конфетти или анимацию при победе. Используйте библиотеку canvas-confetti или напишите свою простую анимацию.

Подсказка

Подключите CDN canvas-confetti и вызовите confetti() при победе.

⭐⭐⭐ Сложное

Турнирная таблица

Добавьте таблицу рекордов, которая показывает 5 лучших результатов (имя + количество попыток).

Подсказка

Сохраняйте массив объектов в localStorage. При новом рекорде добавляйте его в таблицу и сортируйте.

⭐⭐⭐ Сложное

Обратный отсчёт времени

Добавьте таймер — даётся 60 секунд на угадывание. Если время вышло — игра заканчивается.

Подсказка

Используйте setInterval() для отсчёта времени. Не забудьте очистить интервал при новой игре.
